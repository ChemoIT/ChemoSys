```typescript
const entityNameMap = new Map<string, string>()

const lookupPromises: Promise<void>[] = []

// Helper to add lookup
function addLookup(entityType: string, table: string, nameBuilder: (row: any) => string) {
  const ids = entityGroups.get(entityType)
  if (!ids || ids.size === 0) return
  lookupPromises.push(
    supabase
      .from(table)
      .select('id, ...')  // see specific fields below
      .in('id', [...ids])
      .then(({ data }) => {
        for (const row of (data ?? [])) {
          entityNameMap.set(row.id, nameBuilder(row))
        }
      })
  )
}
```
