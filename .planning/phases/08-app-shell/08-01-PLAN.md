After the existing `logout()` function, add:
```typescript
/**
 * logoutApp — signs out the current user and redirects to /chemosys (ChemoSys login).
 * Used by AppLogoutButton in the (app) shell. Distinct from logout() which redirects to /login (admin).
 */
export async function logoutApp(): Promise<never> {
  const supabase = await createClient();
  await supabase.auth.signOut();
```
