3. **Create `src/app/(app)/app/fleet/layout.tsx`** (server component):
   - Import `verifyAppUser`, `getAppNavPermissions` from `@/lib/dal`

   Implementation:
   ```tsx
   export default async function FleetLayout({ children }: { children: React.ReactNode }) {
     // Auth guard — cached, zero extra DB queries
     await verifyAppUser();

     // Get user's app_* permissions
     const permissions = await getAppNavPermissions();

     // Guard: user must have app_fleet permission
     if (!permissions.includes("app_fleet")) {
       redirect("/app");
     }

     // Extract fleet sub-module keys — pass as string[] (not Set — not serializable)
     const fleetPermissions = permissions.filter((k) => k.startsWith("app_fleet_"));

     return (
       <div className="flex flex-1 min-h-0">
         <FleetSidebar permissions={fleetPermissions} />
         <div className="flex-1 overflow-auto p-4 md:p-6">
           {children}
         </div>
       </div>
     );
   }
   ```

   Note: The content area `<div>` gets the `p-4 md:p-6` padding that was removed from the parent layout. This ensures fleet content has padding but the sidebar sits flush.
