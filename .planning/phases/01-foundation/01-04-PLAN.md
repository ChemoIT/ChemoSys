    import { z } from 'zod'

    export const CompanySchema = z.object({
      name: z.string().min(1, 'שם חברה הוא שדה חובה'),
      internal_number: z.string().min(1, 'מספר חברה הוא שדה חובה'),

      contact_name: z.string().optional().or(z.literal('')),
      contact_email: z.string().email('כתובת מייל לא תקינה').optional().or(z.literal('')),

      company_id: z.string().uuid('יש לבחור חברה'),
      parent_dept_id: z.string().uuid().optional().or(z.literal('')).transform(v => v || null),

    4. **Companies Server Actions** (`src/actions/companies.ts`):
    - `'use server'` directive
    - Every action follows the pattern: verifySession -> validate -> mutate -> audit -> revalidate

    `createCompany(prevState: unknown, formData: FormData)`:
    - Call `verifySession()`
    - Parse formData with `CompanySchema.safeParse(Object.fromEntries(formData))`
    - On validation error: return `{ error: parsed.error.flatten().fieldErrors, success: false }`
    - Insert into companies with `created_by: session.userId`
    - Handle unique constraint error (code '23505'): return `{ error: { internal_number: ['מספר חברה כבר קיים'] }, success: false }`
    - Call `writeAuditLog({ userId: session.userId, action: 'INSERT', entityType: 'companies', entityId: data.id, newData: data })`
    - `revalidatePath('/admin/companies')`
    - Return `{ success: true }`

    `updateCompany(id: string, prevState: unknown, formData: FormData)`:
    - Call `verifySession()`
    - Validate with CompanySchema
    - Fetch old data for audit log
    - Update with `updated_by: session.userId`
    - Handle unique constraint error
    - Write audit log with old_data and new_data
    - Revalidate and return

    `softDeleteCompany(id: string)`:
    - Call `verifySession()`
    - Fetch old data for audit log
    - Update `deleted_at` and `updated_by`
    - Write audit log with action: 'DELETE'
    - Revalidate and return

    5. **CompanyForm** (`src/components/admin/companies/CompanyForm.tsx`):
