**1. Create `src/lib/supabase/admin.ts`:**

The admin Supabase client uses the service role key for `auth.admin.*` operations. This file must NEVER be imported in client components.

```typescript
import 'server-only'
import { createClient } from '@supabase/supabase-js'
import type { Database } from '@/types/database'

/**
 * createAdminClient — Supabase client with service_role key.
 * Required for auth.admin.createUser, deleteUser, updateUserById.
 * NEVER import this file in a client component — server-only guard.
 */
export function createAdminClient() {
  return createClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!,
    {
      auth: {
        persistSession: false,
        autoRefreshToken: false,
        detectSessionInUrl: false,
      },
    }
  )
}
```

Verify `server-only` package is available — if not, install it: `npm install server-only`.

**2. Create `src/actions/users.ts`** with seven Server Actions:

**`createUser(prevState, formData)`** — Two-phase atomic operation per research Pattern 2:
- verifySession -> extract employee_id, email, password from formData
- Validate: email is required (use Zod inline or manual check), password min 6 chars, employee_id is UUID
- Step 1: `createAdminClient().auth.admin.createUser({ email, password, email_confirm: true })`
- If auth error: return `{ error: authError.message }` (handles duplicate email)
- Step 2: `supabase.from('users').insert({ auth_user_id: authData.user.id, employee_id, is_blocked: false, created_by: session.userId }).select('id').single()`
- If DB error: ROLLBACK by calling `adminClient.auth.admin.deleteUser(authData.user.id)` then return error
- writeAuditLog on success
- revalidatePath('/admin/users')

**`updateUser(id, prevState, formData)`** — Updates the public.users metadata. For now, the only editable field after creation is notes or linked employee (if allowed). Keep this simple — mainly used if the admin needs to re-link to a different employee or update metadata. Follow updateCompany pattern.

**`softDeleteUser(id)`** — Two-phase per research Pattern 4:
- verifySession -> fetch user record (need auth_user_id)
- Step 1: Soft-delete public.users: `update({ deleted_at: new Date().toISOString() }).eq('id', id)`
- Step 2: Hard-delete auth.users: `adminClient.auth.admin.deleteUser(authUserId)` — frees the email for reuse
- writeAuditLog
- revalidatePath('/admin/users')

**`blockUser(id)`** — Per research Pattern 3:
- verifySession -> fetch user record (need auth_user_id)
- `adminClient.auth.admin.updateUserById(authUserId, { ban_duration: '87600h' })` (10 years)
- `supabase.from('users').update({ is_blocked: true, updated_by: session.userId }).eq('id', id)`
- writeAuditLog with action 'UPDATE', newData includes `{ is_blocked: true }`
- revalidatePath('/admin/users')

**`unblockUser(id)`** — Reverse of blockUser:
- `adminClient.auth.admin.updateUserById(authUserId, { ban_duration: 'none' })`
- `supabase.from('users').update({ is_blocked: false, updated_by: session.userId }).eq('id', id)`
- writeAuditLog, revalidatePath

**`assignTemplate(userId, templateId)`** — Per research Pattern 5:
- verifySession
- Fetch template_permissions for the given templateId
- Delete all non-override user_permissions: `supabase.from('user_permissions').delete().eq('user_id', userId).eq('is_override', false)`
- Insert template permissions as non-override rows: `{ user_id, module_key, level, template_id, is_override: false }`
- Update `users` row to track which template was assigned (store `template_id` if column exists, or just rely on user_permissions tracking)
- writeAuditLog
- revalidatePath('/admin/users')

**`saveUserPermissions(userId, formData)`** — For manual per-user permission override:
- verifySession
- Extract permission data from formData (same `perm_{module_key}` pattern as PermissionMatrixEditor)
- For each module: upsert into user_permissions with `is_override: true, template_id: null`
- Use `supabase.from('user_permissions').upsert({ user_id, module_key, level, is_override: true, template_id: null }, { onConflict: 'user_id,module_key' })`
- writeAuditLog
- revalidatePath('/admin/users')

CRITICAL for all actions: Always use `createAdminClient()` for auth.admin operations and regular `createClient()` for public table operations.

**4. Create `src/app/(admin)/admin/users/page.tsx`:**

Server component:

```typescript
export default async function UsersPage() {
  await verifySession()
  const supabase = await createClient()

  const [usersRes, employeesRes, templatesRes] = await Promise.all([
    supabase
      .from('users')
      .select('*, employees(first_name, last_name, employee_number, email, id_number), user_permissions(module_key, level, is_override, template_id)')
      .is('deleted_at', null)
      .order('created_at', { ascending: false }),
    supabase
      .from('employees')
      .select('id, first_name, last_name, employee_number, email, id_number, companies(name)')
      .is('deleted_at', null)
      .eq('status', 'active')
      .order('last_name'),
    supabase
      .from('role_templates')
      .select('id, name')
      .is('deleted_at', null)
      .order('name'),
  ])

  if (usersRes.error) console.error('[UsersPage]', usersRes.error.message)

  // Compute which employees are already linked to active users
  const linkedEmployeeIds = (usersRes.data ?? []).map(u => u.employee_id)

  return (
    <div className="space-y-6">
      <div className="flex items-center gap-3">
        <h1 className="text-2xl font-bold text-foreground">ניהול יוזרים</h1>
        <Badge variant="secondary">{usersRes.data?.length ?? 0}</Badge>
      </div>
      <UsersTable
        users={usersRes.data ?? []}
        employees={employeesRes.data ?? []}
        linkedEmployeeIds={linkedEmployeeIds}
        templates={templatesRes.data ?? []}
      />
    </div>
  )
}
```
